#!/bin/bash
path=(pwd)
case $1 in 
	1) echo "你选择了安装指定版本"
#local中保存当前可用版本
/usr/local/nginx/sbin/nginx -v &> local
(cat local | grep -o [0-9][0-9].[0-9]) &> local

insall_web="http://nginx.org/download/nginx-${2}.tar.gz"
echo 安装gcc g++依赖库
apt-get install build-essential
apt-get install libtool

echo 安装pcre依赖库............
apt-get update
apt-get install libpcre3 libpcre3-dev

echo 安装zlib依赖库............



apt-get update
apt-get install libpcre3 libpcre3-dev




echo 安装ssl依赖库..............
apt-get install openssl
echo -------------------------------------------------------------------------
echo 安装ssl依赖库..............
apt-get install openssl
echo -------------------------------------------------------------------------
echo 开始安装nginx版本"${2}"

wget -P "${path}" "${install_web}"
if [ $? -eq 0 ]
then
	echo "下载html成功"
else
	echo "下载失败"
	exit 1
fi

echo ---------------------------------------------------------------------------

echo 开始解压
tar zxvf nginx-"${2}".tar.gz
cd "${path}"/nginx-"${2}" || exit
./configure
make
make install
./back
	;;
	2)echo "你选择了更新至最新版本"
#获取本地nginx版本号并存储至b.txt
/usr/local/nginx/sbin/nginx -v &> b.txt
#local中保存当前版本信息
(cat b.txt | grep -o [0-9][0-9].[0-9]) &> local
(cat b.txt | grep -o [0-9][0-9].[0-9]) &> b.txt
#获取官网html
wget https://nginx.org/en/download.html
if [ $? -eq 0 ]
then 
echo "下载html成功"
else
echo "下载失败"
exit 1
fi
(cat download.html | grep -o nginx-1.[0-9]..[0-9]) &> a.txt
sed -n '1p' a.txt &> a.txt
(cat a.txt | grep -o [0-9][0-9].[0-9]) &> a.txt
#a.txt中保存官网最新版本
a=(cat a.txt)
b=(cat b.txt)
if [ "$(expr "${a}" \> "${b}")" -eq 1 ]
then
insall_web="http://nginx.org/download/nginx-1.${a}.tar.gz"
echo 安装gcc g++依赖库
apt-get install build-essential
apt-get install libtool

echo 安装pcre依赖库............
apt-get update
apt-get install libpcre3 libpcre3-dev

echo 安装zlib依赖库............



apt-get update
apt-get install libpcre3 libpcre3-dev

echo 安装ssl依赖库..............
apt-get install openssl
echo -------------------------------------------------------------------------
echo 开始安装nginx版本"${a}"

wget -P "${path}" "${install_web}"
if [ $? -eq 0 ]
then
	echo "下载html成功"
else
	echo "下载失败"
	exit 1
fi

echo ---------------------------------------------------------------------------

echo 开始解压
tar zxvf nginx-1."${a}".tar.gz
cd "${path}"/nginx-1."${a}" || exit
./configure
make
make install 
cd "${path}" || exit
rm download.html
rm nginx-1."${a}".tar.gz
./back


else
echo "当前版本已经是最新版本"
rm download.html
fi

	;;
	*)echo "请在参数1号位输入正确的数字（1或2）"
	;;
esac
